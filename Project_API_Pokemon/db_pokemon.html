💻 Script SQL

CREATE DATABASE pokemon; USE pokemon; CREATE TABLE pokemon ( id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(100) NOT NULL, height INT NOT NULL, weight INT NOT NULL, hp INT NOT NULL, attack INT NOT NULL, defense INT NOT NULL, special_attack INT NOT NULL, special_defense INT NOT NULL, speed INT NOT NULL, types VARCHAR(100), abilities VARCHAR(100), generation INT, region VARCHAR(100), image_path VARCHAR(100) );
